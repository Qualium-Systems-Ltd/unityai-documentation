Introduction

Introducing UnityAI, an innovative plugin that seamlessly integrates Stable Diffusion algorithms, OpenAI's DALL·E, and ChatGPT, empowering users to generate diverse and interactive content within the Unity game development environment. UnityAI combines the power of Stable Diffusion techniques with OpenAI's DALL·E, a state-of-the-art image generation model. With DALL·E, developers can effortlessly generate high-quality and visually appealing images by providing textual prompts. Whether you need realistic environments, character designs, or object models, UnityAI enables you to create stunning visuals that bring your game worlds to life.

Key Features

UnityAI boasts several key features that empower developers to leverage advanced AI capabilities within the Unity game development environment:

  • Stable Diffusion Integration: UnityAI enables users to generate diverse and high-quality content using Stable Diffusion algorithms. From text, images, to audio, developers can create immersive narratives, realistic dialogue, procedural level designs, and more.
  • OpenAI's DALL·E Integration: With UnityAI, developers can seamlessly integrate OpenAI's DALL·E, a powerful image generation model. By providing textual prompts, users can effortlessly generate visually stunning and realistic graphics for environments, characters, objects, and more.
  • ChatGPT Integration: UnityAI incorporates ChatGPT, an advanced language model developed by OpenAI, allowing for interactive and dynamic conversations within Unity projects. This feature enables developers to create engaging interactions between players and non-player characters (NPCs), enhancing the overall gameplay experience.
  • Easy Integration: UnityAI provides a user-friendly interface and easy-to-use API, making it straightforward to integrate AI functionalities into Unity projects. Developers can quickly implement AI-driven behaviors, decision-making systems, and content generation with minimal effort.

API Settings

API Settings is a scriptable object used to store and manage API settings, such as API key and API endpoint, in a project.

API Settings (Scriptable Object)

APISettings - is a scriptable object which contains API key and API endpoint for requests. You can use default settings (you can find them in the "Assets/UnityAI/Settings" folder of your project.), but you have to set your personal API key, or you can create a new one.

Constructor

public APISettings(string settingsName, string apiAddress, string apiKey)

string settingsName Settings name
string apiAddress Endpoint address
string apiKey API Key
Properties
string settingsName Settings name
string apiAddress Endpoint address
string apiKey API Key
Creation

In "Project" overview press RMB, in drop-down menu select "Create", then "UnityAI", then "API Settings". For created object set API key and API address.

Preview

using UnityEngine;

[CreateAssetMenu (menuName = "UnityAI/API Settings", fileName = "New API Settings", order = -1)]
public class APISettings : ScriptableObject
{
  public string settingsName;
  public string apiAddress;
  public string apiKey;

  public APISettings(string settingsName, string apiAddress, string apiKey)
  {
    this.settingsName = settingsName;
    this.apiAddress = apiAddress;
    this.apiKey = apiKey;
  }
}

Stable Diffusion (Misc)

Stable Diffusion Balance

Get the credit balance of the account/organization associated with the API key.

Example:

using QS.StableDiffusion;
using UnityEngine;

public class ExampleClass : MonoBehaviour
{
	public APISettings sdBalanceSettings;
	public float credits;

	public void Request()
	{
		StartCoroutine(RequestsHandler.GetRequest(sdBalanceSettings, ResponseHandler));
	}

	public void ResponseHandler(string responseJson)
	{
		BalanceResponse response = BalanceResponse.FromJSON(responseJson);
		credits = response.credits;
	}
}

Example

using System;
using System.Collections.Generic;
using UnityEngine;
using static QS.StableDiffusion.SDImage;

public class ExampleClass : MonoBehaviour
{
	public MeshRenderer imgDisplayer;
	public List<Texture2D> generatedImages;
	public APISettings sdTextToImgSettings;
	public TextPrompt[] textPrompts;
	public ClipGuidance clipGuidance = ClipGuidance.NONE;
	public Sampler sampler = Sampler.K_LMS;
	public Dimension height = Dimension._512;
	public Dimension width = Dimension._512;
	[Range(1, 10)] public int samples = 1;
	[Range(0, 35)] public float cfgScale = 4;
	[Range(10, 150)] public int steps = 30;
	[Range (0, 4294967295)] public uint seed = 0;

	public void Generate()
	{
		SDImage.TextToImg request = new SDImage.TextToImg(textPrompts, cfgScale, height, width, samples, steps, seed, clipGuidance, sampler);
		StartCoroutine(RequestsHandler.PostRequest(request.GenerateBody(), sdTextToImgSettings, ImageResponseHandler));
	}

	public void ImageResponseHandler(string responseJson)
	{
		Response response = JsonUtility.FromJson<Response>(responseJson);
		for (int i = 0; i < response.artifacts.Length; i++)
		{
			string imgB64 = response.artifacts[i].base64;
			string seed = response.artifacts[i].seed;
			if (imgB64 != null) Utils.Base64ToImg(imgB64, TextureHandler, seed);
		}
	}

	public void TextureHandler(Texture2D texture, string imgName)
	{
		EditorUtils.TextureHandler(texture, imgName, generatedImages, imgDisplayer);
	}
}

using System.Collections.Generic;
using UnityEngine;
using static QS.OpenAI.OAIImage;

public class ExampleClass : MonoBehaviour
{
	public MeshRenderer imgDisplayer;
	public List<Texture2D> generatedImages;
	public APISettings oaiImgSettings;

		Request request = new Request(imgPrompt, samples, imgSize, responseFormat);
		StartCoroutine(RequestsHandler.PostRequest(request.GenerateBody(), oaiImgSettings, ImageResponseHandler));
	}

	public void ImageResponseHandler(string responseJson)
	{
		Response response = JsonUtility.FromJson<Response>(responseJson);
		string imgName = response.created.ToString();
		for (int i = 0; i < response.data.Length; i++)
		{
			string imgUrl = response.data[i].url;
			string imgB64 = response.data[i].b64_json;
			if (imgUrl != null) StartCoroutine(Utils.UrlToImg(imgUrl, TextureHandler));
			if (imgB64 != null) Utils.Base64ToImg(imgB64, TextureHandler, imgName);
		}
	}

	public void TextureHandler(Texture2D texture, string imgName)
	{
		EditorUtils.TextureHandler(texture, imgName, generatedImages, imgDisplayer);

Example

using System.Collections.Generic;
using UnityEngine;
using static QS.OpenAI.OAIChat;

public class ExampleClass : MonoBehaviour
{
	public APISettings oaiChatSettings;
	public string chatRequest;
	public Model chatModel = Model.ChatGPT3_5_turbo;
	[Range(0, 2f)] public float temperature = 1;
	[Range(0, 4096)] public int maxTokens = 1024;

	public List<Message> messagesHistory = new List<Message>();

	private void Start()
	{
		Request();
	}

	public void Request()
	{
		Message msg = new Message(Role.User, chatRequest);
		messagesHistory.Add(msg);
		Request request = new Request(chatModel, messagesHistory.ToArray(), temperature, maxTokens);
		StartCoroutine(RequestsHandler.PostRequest(request.GenerateBody(), oaiChatSettings, ChatResponseHandler));
	}

	public void ChatResponseHandler(string responseJson)
	{
		Response response = JsonUtility.FromJson<Response>(responseJson);
		Message msg = new Message(Role.Assistant, response.choices[0].message.content);
		messagesHistory.Add(msg);
	}
}
